########################################################################
# Ship Market - Scripted Effects (v0.7)
# Target: Stellaris 4.4.3 "Pegasus".
# MOTHBALL and LISTING are now SEPARATE:
#   * A fleet is MOTHBALLED (inert) via the sm_mothballed flag + sm_mothballed_fleet
#     modifier. Stats (power, ship counts) are cached on the fleet AT MOTHBALL TIME,
#     while it still has real fleet_power.
#   * A fleet is LISTED for sale via the sm_market_listing flag + price variable.
#     Listing requires the fleet already be mothballed.
# Recommission timings: leaving the MARKET (retract or sale) = 10 days; waking a
# stored fleet from mothball (un-mothball) = 360 days.
# UNLIMITED listings/mothballs: no fixed global slots. Menus snapshot up to 10 at
# a time (sm_bcand_* / sm_rcand_* / sm_cand_* / sm_mocand_* / sm_ucand_*).
# Scripted-effect call depth is kept <= 5 (engine limit).
########################################################################

# ----- Mothball-count + seller modifier (country scope, root/this = seller) --
# Recompute the count of MOTHBALLED fleets (listed or not) from flags (no drift)
# and (re)apply the upkeep discount scaled by it. Called whenever a seller's
# mothballed/listed fleets change. (Variable name kept for compatibility; it now
# counts all mothballed fleets, not just listed ones.)
sm_refresh_seller_mothball_modifier = {
	set_variable = { which = sm_active_ship_market_listings value = 0 }
	every_owned_fleet = {
		limit = { has_fleet_flag = sm_mothballed }
		prev = { change_variable = { which = sm_active_ship_market_listings value = 1 } }
	}
	# v0.9.30 EXPLOIT FIX: cap the stack at 10 fleets (-20%, civic -30%). The
	# count is per FLEET, so unbounded stacking let 1-corvette splits zero out
	# the whole navy's upkeep (34 mothballed corvettes = -68%).
	if = {
		limit = { check_variable = { which = sm_active_ship_market_listings value > 10 } }
		set_variable = { which = sm_active_ship_market_listings value = 10 }
	}
	remove_modifier = sm_mothball_upkeep_discount
	# v0.9.29: dealer civics get a x1.5 stronger inventory-holding discount.
	if = {
		limit = {
			check_variable = { which = sm_active_ship_market_listings value > 0 }
			OR = {
				has_valid_civic = civic_sm_arms_consortium
				has_valid_civic = civic_sm_fleet_chandlers
			}
		}
		add_modifier = {
			modifier = sm_mothball_upkeep_discount
			multiplier = value:sm_seller_listing_count_value_civic
			days = -1
		}
	}
	else_if = {
		limit = { check_variable = { which = sm_active_ship_market_listings value > 0 } }
		add_modifier = {
			modifier = sm_mothball_upkeep_discount
			multiplier = value:sm_seller_listing_count_value
			days = -1
		}
	}
}

# ----- Mothball (fleet scope, root = owner) ----------------------------------
# Cache stats FIRST (while the fleet still has real power), then flag + apply the
# pure stat modifier. Reverses cleanly. No set_disabled / set_fleet_settings.
sm_mothball_fleet = {
	clear_orders = yes
	sm_capture_fleet_stats = yes
	set_fleet_flag = sm_mothballed
	add_modifier = { modifier = sm_mothballed_fleet days = -1 }
}

# FLEET SCOPE. Drop the cached stat variables (shared by clear-mothball paths).
sm_clear_fleet_stats = {
	clear_variable = sm_fp
	clear_variable = sm_nships
	clear_variable = sm_n_corv
	clear_variable = sm_n_frig
	clear_variable = sm_n_dest
	clear_variable = sm_n_cru
	clear_variable = sm_n_bb
	clear_variable = sm_n_tit
	clear_variable = sm_cost_alloys
	clear_variable = sm_build_value
	clear_variable = sm_quality
}

# FLEET SCOPE. Remove the MOTHBALL layer (flag + modifier + cached stats). No
# recommission timer of its own — callers add sm_reactivating_fleet as needed.
sm_clear_mothball = {
	remove_fleet_flag = sm_mothballed
	remove_modifier = sm_mothballed_fleet
	sm_clear_fleet_stats = yes
}

# FLEET SCOPE (root = owner). Wake a stored, UNLISTED fleet back to operational:
# drop the mothball layer, then begin a 360-day recommission (timed
# sm_reactivating_fleet auto-expires -> normal fleet). Caller refreshes the modifier.
# sm_react_total/left drive the in-menu "~N days left" estimate (see sm_tick_reactivating).
sm_unmothball_fleet = {
	sm_clear_mothball = yes
	# v0.9.29: dealer civics run leaner yards - recommission 360 -> 270 days.
	if = {
		limit = {
			root = {
				OR = {
					has_valid_civic = civic_sm_arms_consortium
					has_valid_civic = civic_sm_fleet_chandlers
				}
			}
		}
		add_modifier = { modifier = sm_reactivating_fleet days = 270 }
		set_variable = { which = sm_react_total value = 270 }
		set_variable = { which = sm_react_left  value = 270 }
	}
	else = {
		add_modifier = { modifier = sm_reactivating_fleet days = 360 }
		set_variable = { which = sm_react_total value = 360 }
		set_variable = { which = sm_react_left  value = 360 }
	}
}

# COUNTRY SCOPE (owner). Monthly: tick down the displayed recommission estimate on
# each fleet still recommissioning. The sm_reactivating_fleet modifier is the exact,
# engine-managed countdown (visible in its tooltip); sm_react_left is only a coarse
# ~monthly estimate shown in the Recommissioning Status menu.
sm_tick_reactivating = {
	every_owned_fleet = {
		limit = { has_modifier = sm_reactivating_fleet }
		change_variable = { which = sm_react_left value = -30 }
		if = {
			limit = { check_variable = { which = sm_react_left value < 0 } }
			set_variable = { which = sm_react_left value = 0 }
		}
	}
}

# FLEET SCOPE. Store fleet power / ship counts on the fleet for the inspect UI.
sm_capture_fleet_stats = {
	save_event_target_as = sm_stat_fleet
	set_variable = { which = sm_fp value = trigger:fleet_power }
	set_variable = { which = sm_nships value = trigger:num_ships }
	set_variable = { which = sm_n_corv value = 0 }
	set_variable = { which = sm_n_frig value = 0 }
	set_variable = { which = sm_n_dest value = 0 }
	set_variable = { which = sm_n_cru value = 0 }
	set_variable = { which = sm_n_bb value = 0 }
	set_variable = { which = sm_n_tit value = 0 }
	every_owned_ship = { limit = { is_ship_size = corvette }   event_target:sm_stat_fleet = { change_variable = { which = sm_n_corv value = 1 } } }
	every_owned_ship = { limit = { is_ship_size = frigate }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_frig value = 1 } } }
	every_owned_ship = { limit = { is_ship_size = destroyer }  event_target:sm_stat_fleet = { change_variable = { which = sm_n_dest value = 1 } } }
	every_owned_ship = { limit = { is_ship_size = cruiser }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_cru value = 1 } } }
	every_owned_ship = { limit = { is_ship_size = battleship } event_target:sm_stat_fleet = { change_variable = { which = sm_n_bb value = 1 } } }
	every_owned_ship = { limit = { is_ship_size = titan }      event_target:sm_stat_fleet = { change_variable = { which = sm_n_tit value = 1 } } }
	# Store the alloy build-cost estimate now (counts above are live in sm_stat_fleet),
	# so the live-market price recommendation can read it at listing time.
	set_variable = { which = sm_cost_alloys value = value:sm_fleet_alloys }
	# Estimated TRADE build cost, counted via count_owned_ship restricted to this fleet
	# (sm_stat_fleet). Captured AT MOTHBALL and stored; pricing reads sm_build_value.
	set_variable = { which = sm_build_value value = value:sm_fleet_build_value }
	# v0.9.40: sm_apply_market_value INLINED here (not called). The extra scripted-
	# effect frames pushed the AI-deal chains (sm_ai_force_deal_now -> sm_ai_send_* ->
	# sm_mothball_fleet -> here) past the engine's post-init nesting cap -> effects
	# left uninitialized -> CTD at new-game init (4.4.5). The sm_ensure_cost_alloys
	# backfill is skipped on this path: counts + sm_cost_alloys are fresh (just set
	# above). KEEP IN SYNC with sm_apply_market_value below.
	# Direct `owner` scope refs, NOT an event target: on 4.4.5 a target saved
	# inside the owner block is already undefined right after the block exits
	# (confirmed via error.log), which silently killed the live-market price.
	if = {
		limit = { exists = owner }
		owner = { set_variable = { which = sm_mkt_tmp value = value:sm_fleet_market_value } }
		if = {
			limit = { owner = { check_variable = { which = sm_mkt_tmp value >= 1 } } }
			set_variable = { which = sm_build_value value = owner.sm_mkt_tmp }
		}
		owner = { clear_variable = sm_mkt_tmp }
	}
	# v0.9.31 QUALITY appraisal: absolute per-ship costs are NOT readable in 4.4
	# (ship_size_cost_resource_percent returns composition fractions only), so we
	# proxy "what it actually took to build" with the fleet's POWER at mothball
	# vs a tier-1 reference power for the same hull mix, clamped 0.5..3.0. A
	# fully-teched battleship line appraises ~2-3x a starter one. Captured HERE
	# (power is real pre-modifier) and reapplied at every re-appraisal.
	set_variable = { which = sm_ref_power value = 0 }
	set_variable = { which = sm_qtmp value = sm_n_corv }  multiply_variable = { which = sm_qtmp value = 40 }   change_variable = { which = sm_ref_power value = sm_qtmp }
	set_variable = { which = sm_qtmp value = sm_n_frig }  multiply_variable = { which = sm_qtmp value = 60 }   change_variable = { which = sm_ref_power value = sm_qtmp }
	set_variable = { which = sm_qtmp value = sm_n_dest }  multiply_variable = { which = sm_qtmp value = 100 }  change_variable = { which = sm_ref_power value = sm_qtmp }
	set_variable = { which = sm_qtmp value = sm_n_cru }   multiply_variable = { which = sm_qtmp value = 250 }  change_variable = { which = sm_ref_power value = sm_qtmp }
	set_variable = { which = sm_qtmp value = sm_n_bb }    multiply_variable = { which = sm_qtmp value = 650 }  change_variable = { which = sm_ref_power value = sm_qtmp }
	set_variable = { which = sm_qtmp value = sm_n_tit }   multiply_variable = { which = sm_qtmp value = 3000 } change_variable = { which = sm_ref_power value = sm_qtmp }
	clear_variable = sm_qtmp
	set_variable = { which = sm_quality value = 1 }
	if = {
		limit = {
			check_variable = { which = sm_ref_power value > 0 }
			check_variable = { which = sm_fp value > 0 }
		}
		set_variable = { which = sm_quality value = sm_fp }
		divide_variable = { which = sm_quality value = sm_ref_power }
		if = { limit = { check_variable = { which = sm_quality value < 0.5 } } set_variable = { which = sm_quality value = 0.5 } }
		if = { limit = { check_variable = { which = sm_quality value > 3 } }   set_variable = { which = sm_quality value = 3 } }
	}
	clear_variable = sm_ref_power
	# v0.9.40: sm_apply_build_quality INLINED (same nesting-cap reason as the
	# market-value block above). KEEP IN SYNC with sm_apply_build_quality below.
	if = {
		limit = { check_variable = { which = sm_quality value > 0 } }
		multiply_variable = { which = sm_build_value value = sm_quality }
	}
	if = {
		limit = { exists = owner  owner = { has_valid_civic = civic_sm_assembly_line_shipwrights } }
		multiply_variable = { which = sm_build_value value = 1.2 }
	}
}

# FLEET SCOPE. Scale a freshly recomputed sm_build_value by the mothball-time
# QUALITY multiplier (v0.9.31). Safe no-op when quality was never captured.
sm_apply_build_quality = {
	if = {
		limit = { check_variable = { which = sm_quality value > 0 } }
		multiply_variable = { which = sm_build_value value = sm_quality }
	}
	# v0.9.39 Shipwright appraisal: hulls from Assembly-Line yards are catalogue
	# products and appraise +20% (order-book pricing - the civic's seller-side
	# upside; the -20% fire-rate malus is unchanged).
	if = {
		limit = { exists = owner  owner = { has_valid_civic = civic_sm_assembly_line_shipwrights } }
		multiply_variable = { which = sm_build_value value = 1.2 }
	}
}

# FLEET SCOPE. v0.9.39 LIVE-MARKET APPRAISAL: replace the static-table
# sm_build_value with the trade cost of buying this fleet's alloys on the
# market right now (sm_fleet_market_value). Falls back silently to the static
# value already in sm_build_value when the market read returns 0. Call AFTER
# the static recompute and BEFORE sm_apply_build_quality.
sm_apply_market_value = {
	save_event_target_as = sm_stat_fleet
	# v0.9.40: sm_ensure_cost_alloys INLINED - the call put the AI-deal chains
	# (sm_ai_force_deal_now -> sm_ai_* -> sm_set_recommended_price -> here) at the
	# engine's post-init nesting cap. KEEP IN SYNC with sm_ensure_cost_alloys below.
	if = {
		limit = { NOT = { check_variable = { which = sm_cost_alloys value > 0 } } }
		set_variable = { which = sm_n_corv value = 0 }
		set_variable = { which = sm_n_frig value = 0 }
		set_variable = { which = sm_n_dest value = 0 }
		set_variable = { which = sm_n_cru value = 0 }
		set_variable = { which = sm_n_bb value = 0 }
		set_variable = { which = sm_n_tit value = 0 }
		every_owned_ship = { limit = { is_ship_size = corvette }   event_target:sm_stat_fleet = { change_variable = { which = sm_n_corv value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = frigate }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_frig value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = destroyer }  event_target:sm_stat_fleet = { change_variable = { which = sm_n_dest value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = cruiser }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_cru value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = battleship } event_target:sm_stat_fleet = { change_variable = { which = sm_n_bb value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = titan }      event_target:sm_stat_fleet = { change_variable = { which = sm_n_tit value = 1 } } }
		set_variable = { which = sm_cost_alloys value = value:sm_fleet_alloys }
	}
	# Direct `owner` scope refs, NOT an event target: on 4.4.5 a target saved
	# inside the owner block is already undefined right after the block exits
	# (confirmed via error.log). KEEP IN SYNC with the inline copy in
	# sm_capture_fleet_stats above.
	if = {
		limit = { exists = owner }
		owner = { set_variable = { which = sm_mkt_tmp value = value:sm_fleet_market_value } }
		if = {
			limit = { owner = { check_variable = { which = sm_mkt_tmp value >= 1 } } }
			set_variable = { which = sm_build_value value = owner.sm_mkt_tmp }
		}
		owner = { clear_variable = sm_mkt_tmp }
	}
}

# FLEET SCOPE. Backfill the alloy estimate for a fleet that lacks it (legacy/desynced
# fleets mothballed before sm_cost_alloys existed). Recomputes from a fresh ship count
# only when missing; does NOT touch sm_fp (a mothballed fleet reports 0 power, and we
# want to keep the power captured while it was active). Safe to call any time.
sm_ensure_cost_alloys = {
	if = {
		limit = { NOT = { check_variable = { which = sm_cost_alloys value > 0 } } }
		save_event_target_as = sm_stat_fleet
		set_variable = { which = sm_n_corv value = 0 }
		set_variable = { which = sm_n_frig value = 0 }
		set_variable = { which = sm_n_dest value = 0 }
		set_variable = { which = sm_n_cru value = 0 }
		set_variable = { which = sm_n_bb value = 0 }
		set_variable = { which = sm_n_tit value = 0 }
		every_owned_ship = { limit = { is_ship_size = corvette }   event_target:sm_stat_fleet = { change_variable = { which = sm_n_corv value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = frigate }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_frig value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = destroyer }  event_target:sm_stat_fleet = { change_variable = { which = sm_n_dest value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = cruiser }    event_target:sm_stat_fleet = { change_variable = { which = sm_n_cru value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = battleship } event_target:sm_stat_fleet = { change_variable = { which = sm_n_bb value = 1 } } }
		every_owned_ship = { limit = { is_ship_size = titan }      event_target:sm_stat_fleet = { change_variable = { which = sm_n_tit value = 1 } } }
		set_variable = { which = sm_cost_alloys value = value:sm_fleet_alloys }
	}
}

# COUNTRY SCOPE. Self-heal the mothball state on every owned fleet: the sm_mothballed
# flag and the sm_mothballed_fleet modifier must travel together, but save/load or a
# legacy data model can desync them (e.g. flag kept, modifier dropped on load — the
# fleet can still be reactivated but isn't shown/treated as mothballed). Also backfills
# the alloy estimate so the price recommendation is never 0. Runs on the monthly pulse
# and whenever a management menu opens.
sm_reconcile_mothball_state = {
	every_owned_fleet = {
		limit = { has_fleet_flag = sm_mothballed  NOT = { has_modifier = sm_reactivating_fleet } }
		if = {
			limit = { NOT = { has_modifier = sm_mothballed_fleet } }
			add_modifier = { modifier = sm_mothballed_fleet days = -1 }
		}
		sm_ensure_cost_alloys = yes
	}
	every_owned_fleet = {
		limit = { has_modifier = sm_mothballed_fleet  NOT = { has_fleet_flag = sm_mothballed } }
		set_fleet_flag = sm_mothballed
	}
}

# COUNTRY SCOPE (seller, root). Set the working price to the LIVE-MARKET recommended
# value for event_target:sm_pending_fleet (the trade-credit cost to buy the fleet's alloy
# value at the current market price). Evaluated in the seller's market scope.
sm_set_recommended_price = {
	# v0.9.30: re-appraise the fleet first (see sm_list_fleet).
	if = {
		limit = { exists = event_target:sm_pending_fleet }
		event_target:sm_pending_fleet = {
			save_event_target_as = sm_stat_fleet
			set_variable = { which = sm_build_value value = value:sm_fleet_build_value }
			sm_apply_market_value = yes
			sm_apply_build_quality = yes
		}
	}
	# Primary: estimated build cost x1.5 (sm_recommended_price reads stored sm_build_value).
	set_variable = { which = sm_working_price value = value:sm_recommended_price }
	# Fallback: fleet-power valuation if the build estimate is unavailable/zero
	# (trade scale: ~1 credit per point of fleet power).
	if = {
		limit = { check_variable = { which = sm_working_price value < 1 } }
		set_variable = { which = sm_working_price value = event_target:sm_pending_fleet.sm_fp }
	}
	# Floor so a sale price is never 0.
	if = {
		limit = { check_variable = { which = sm_working_price value < 50 } }
		set_variable = { which = sm_working_price value = 50 }
	}
	sm_clamp_working_price = yes
}

# COUNTRY SCOPE (seller). Keep the working price within [0, 50,000] - the vanilla
# trade STOCKPILE CAP: no buyer can ever hold more, so a higher ask is unbuyable.
sm_clamp_working_price = {
	if = { limit = { check_variable = { which = sm_working_price value < 0 } }      set_variable = { which = sm_working_price value = 0 } }
	if = { limit = { check_variable = { which = sm_working_price value > 50000 } } set_variable = { which = sm_working_price value = 50000 } }
}

# ----- List / unlist (fleet scope) -------------------------------------------
# FLEET SCOPE. root = seller. Mark this ALREADY-MOTHBALLED fleet as a listing at
# the working price. Stats were cached at mothball time, so we only add the market
# flag + price here.
sm_list_fleet = {
	# v0.9.30 EXPLOIT FIX: re-appraise NOW. Ship COUNTS stay readable under the
	# mothball modifier, and the cached sm_build_value could be gamed by splitting
	# ships out after mothballing (sell a husk at full-fleet "fair" price).
	save_event_target_as = sm_stat_fleet
	set_variable = { which = sm_build_value value = value:sm_fleet_build_value }
	sm_apply_market_value = yes
	sm_apply_build_quality = yes
	set_fleet_flag = sm_market_listing
	set_fleet_flag = sm_market_locked
	set_variable = { which = sm_price_trade value = root.sm_working_price }
	# Deal quality for AI buyers: asking price as a PERCENT of the estimated build
	# value (sm_build_value). <=100 is a bargain; >150 is steep. 999 if value unknown.
	if = {
		limit = { check_variable = { which = sm_build_value value >= 1 } }
		set_variable = { which = sm_deal_pct value = sm_price_trade }
		divide_variable = { which = sm_deal_pct value = sm_build_value }
		multiply_variable = { which = sm_deal_pct value = 100 }
	}
	else = { set_variable = { which = sm_deal_pct value = 999 } }
	# Export Subsidies: the listing READS ~30% cheaper to AI evaluators (perception
	# only - the price paid is unchanged). Baked at listing time. v0.9.39: 0.85 ->
	# 0.7 so a fair ask (~105%) actually crosses the <75 bargain band.
	if = {
		limit = { exists = owner  owner = { has_edict = sm_export_subsidies } }
		multiply_variable = { which = sm_deal_pct value = 0.7 }
	}
	owner = {
		set_country_flag = sm_just_listed
		sm_refresh_seller_mothball_modifier = yes
	}
}

# FLEET SCOPE. Remove ONLY the MARKET layer (flags + price). The fleet stays
# MOTHBALLED — retracting a listing returns it to plain mothball storage, NOT to
# active service. To make it operational the owner must un-mothball it afterward
# (360-day recommission). Used by retract. Does NOT refresh the seller's modifier
# — callers do that (so bulk cancel doesn't refresh per-fleet). The fleet remains
# counted as mothballed, so the upkeep discount is unchanged by a retract.
sm_unlist_fleet = {
	remove_fleet_flag = sm_market_listing
	remove_fleet_flag = sm_market_locked
	clear_variable = sm_price_trade
	clear_variable = sm_deal_pct
}

# FLEET SCOPE. Force a fleet fully off the market AND out of mothball, then begin a
# 10-day recommission to operational. Used by a SALE (the buyer receives a fleet
# that commissions in 10 days) and by debug recovery. NOT used by a seller's
# retract — that intentionally leaves the fleet mothballed.
sm_force_reactivate_fleet = {
	sm_unlist_fleet = yes
	sm_clear_mothball = yes
	add_modifier = { modifier = sm_reactivating_fleet days = 10 }
	set_variable = { which = sm_react_total value = 10 }
	set_variable = { which = sm_react_left  value = 10 }
}

# COUNTRY SCOPE (seller, root). List the picked pending fleet, if still eligible.
sm_list_pending_fleet = {
	if = {
		limit = {
			exists = event_target:sm_pending_fleet
			event_target:sm_pending_fleet = { sm_fleet_can_be_listed = yes }
		}
		event_target:sm_pending_fleet = { sm_list_fleet = yes }
	}
}

# COUNTRY SCOPE (seller, root). Retract every fleet I currently have listed.
sm_retract_all_my_listings = {
	every_owned_fleet = {
		limit = { has_fleet_flag = sm_market_listing }
		sm_unlist_fleet = yes
	}
	sm_refresh_seller_mothball_modifier = yes
}

# COUNTRY SCOPE (root = owner). Mothball EVERY eligible active fleet at once.
# Designed for "build a fleet of fleets, mothball them all, list later".
sm_mothball_all_eligible = {
	every_owned_fleet = {
		limit = { sm_fleet_can_be_mothballed = yes }
		sm_mothball_fleet = yes
	}

# COUNTRY SCOPE. root = buyer. Buy the inspected fleet (event_target:sm_inspect_fleet).
# Self-guarding: re-validates the listing at execution time and fires the result.
sm_buy_fleet = {
	if = {
		limit = {
			exists = event_target:sm_inspect_fleet
			event_target:sm_inspect_fleet = { sm_fleet_is_buyable = yes }
			resource_stockpile_compare = { resource = trade value >= value:sm_inspect_price }
		}
		set_variable = { which = sm_last_market_value value = value:sm_inspect_price }
		add_resource = { trade = -1 mult = value:sm_inspect_price }
		# Pay + notify the seller (the fleet's current owner).
		event_target:sm_inspect_fleet = {
			owner = {
				save_event_target_as = sm_sale_seller
				set_variable = { which = sm_last_market_value value = value:sm_inspect_price }
				add_resource = { trade = 1 mult = value:sm_inspect_price }
				country_event = { id = sm_market.230 }
			}
			# Off-market + out of mothball + 10-day recommission, then transfer
			# ownership AND control (buyer gets an operational fleet in 10 days).
			# v0.9.25: also strip AI-deal bookkeeping so it can't follow the fleet
			# to its new owner (AI board listings carry sm_ai_offered + a shelf life).
			remove_fleet_flag = sm_ai_offered
			remove_fleet_flag = sm_ai_wait
			clear_variable = sm_ai_list_ttl
			sm_force_reactivate_fleet = yes
			set_owner = root
			set_controller = root
			clear_orders = yes
			# Deliver instantly: teleport the fleet to the buyer's capital system.
			if = {
				limit = { exists = root.capital_scope }
				set_location = { target = root.capital_scope  distance = 0 }
			}
		}
		# Refresh the (former) seller's listing modifier now their count dropped.
		event_target:sm_sale_seller = { sm_refresh_seller_mothball_modifier = yes }
		log = "SM_DEAL,board_sale,buyer=[Root.GetName],seller=[sm_sale_seller.GetName],price=[Root.GetShipMarketLastValue]"
		country_event = { id = sm_market.231 }
	}
	else = {
		country_event = { id = sm_market.242 }
	}
}

# ----- AI: list a surplus fleet for sale -------------------------------------
# COUNTRY SCOPE (root = AI seller). Pick one surplus military fleet, unassign its
# admiral, mothball it, and list it at a modest price (alloy market value x2).
# Guarded so the AI keeps at least one fleet and holds at most 3 listings at once.
# Called from sm_market.910 (monthly, at peace).
sm_ai_list_surplus_fleet = {
	# Count current listings (cap a few per empire) and spare sellable fleets.
	set_variable = { which = sm_ai_listing_count value = 0 }
	every_owned_fleet = {
		limit = { has_fleet_flag = sm_market_listing }
		prev = { change_variable = { which = sm_ai_listing_count value = 1 } }
	}
	set_variable = { which = sm_ai_eligible_count value = 0 }
	every_owned_fleet = {
		limit = { sm_fleet_ai_sellable = yes }
		prev = { change_variable = { which = sm_ai_eligible_count value = 1 } }
	}
	if = {
		# Up to 3 listings at once, and only while keeping at least one spare fleet.
		limit = {
			check_variable = { which = sm_ai_listing_count value < 3 }
			check_variable = { which = sm_ai_eligible_count value >= 2 }
		}
		random_owned_fleet = {
			limit = { sm_fleet_ai_sellable = yes }
			save_event_target_as = sm_ai_sell_fleet
		}
		if = {
			limit = { exists = event_target:sm_ai_sell_fleet }
			event_target:sm_ai_sell_fleet = {
				if = { limit = { exists = leader } leader = { unassign_leader = this } }
				sm_mothball_fleet = yes
				save_event_target_as = sm_pending_fleet
			}
			# Price via the SAME robust path as the player's suggestion
			# (sm_set_recommended_price: ensures the alloy estimate, prices at market
			# x3 with a guaranteed non-zero fallback), then trim to a modest ~x2 for
			# AI sellers. Reusing it guarantees the AI is never priced at 0.
			sm_set_recommended_price = yes
			multiply_variable = { which = sm_working_price value = 0.67 }
			sm_clamp_working_price = yes
			event_target:sm_ai_sell_fleet = { sm_list_fleet = yes }
			remove_country_flag = sm_just_listed
		}
	}
}

# FLEET SCOPE. Drop the sale-offer reservation (flag + price/deal vars). The fleet
# stays MOTHBALLED. Shared by accept/decline/withdraw paths.
sm_clear_sale_offer_fleet = {
	remove_fleet_flag = sm_offer_pending
	clear_variable = sm_price_trade
	clear_variable = sm_deal_pct
}

# COUNTRY SCOPE (seller, root). Send a sale offer for event_target:sm_pending_fleet
# at root.sm_working_price to event_target:sm_offer_recipient. Reserves the fleet,
# publishes the offer globals, and fires the recipient's decision event. Sets the
# sm_sale_offer_sent flag so the caller can confirm it went out.
sm_send_sale_offer = {
	remove_country_flag = sm_sale_offer_sent
	if = {
		limit = {
			exists = event_target:sm_pending_fleet
			exists = event_target:sm_offer_recipient
			event_target:sm_pending_fleet = { sm_fleet_offerable_sale = yes }
			sm_deal_slot_free = yes								# a concurrent slot is free
			NOT = { event_target:sm_offer_recipient = { has_country_flag = sm_pending_player_offer } }	# one offer per empire
		}
		event_target:sm_pending_fleet = {
		# v0.9.30 EXPLOIT FIX: re-appraise NOW. Ship COUNTS stay readable under the
		# mothball modifier, and the cached sm_build_value could be gamed by splitting
		# ships out after mothballing (sell a husk at full-fleet "fair" price).
		save_event_target_as = sm_stat_fleet
		set_variable = { which = sm_build_value value = value:sm_fleet_build_value }
		set_fleet_flag = sm_offer_pending
			set_variable = { which = sm_price_trade value = root.sm_working_price }
			# Deal quality for the AI recipient: asking price as a PERCENT of the
			# estimated build value. <=100 bargain, >150 steep, 999 if value unknown.
			if = {
				limit = { check_variable = { which = sm_build_value value >= 1 } }
				set_variable = { which = sm_deal_pct value = sm_price_trade }
				divide_variable = { which = sm_deal_pct value = sm_build_value }
				multiply_variable = { which = sm_deal_pct value = 100 }
			}
			else = { set_variable = { which = sm_deal_pct value = 999 } }
			# v0.9.29 Export Subsidies: offer READS ~15% cheaper (perception only).
			if = {
				limit = { root = { has_edict = sm_export_subsidies } }
				multiply_variable = { which = sm_deal_pct value = 0.7 }	# v0.9.39: was 0.85
			}
			save_global_event_target_as = sm_deal_new_fleet		# hand to the slot allocator
		}
		event_target:sm_offer_recipient = {
			# Claim a concurrent slot (type 1 = sale), show an "O" row, and fire the
			# recipient's decision after a visible pending window. Slots let offers to
			# multiple empires coexist without clobbering (sm_deal_slots.txt).
			set_country_flag = sm_pending_player_offer
			set_variable = { which = sm_my_deal_type value = 1 }
			set_variable = { which = sm_offer_ttl value = 30 }		# auto-decline after ~30 days if unanswered (sm_offer.900)
			sm_alloc_deal_slot = yes
			country_event = { id = sm_offer.100  days = 3 }
		}
		set_country_flag = sm_sale_offer_sent
	}
}

# COUNTRY SCOPE (buyer, root). Fulfil an accepted sale offer: re-validate, pay the
# seller, transfer ownership AND control, deliver to the buyer's capital, and start
# a 10-day recommission. Sets sm_sale_bought on success so the caller can branch.
sm_accept_sale_offer = {
	remove_country_flag = sm_sale_bought
	sm_load_deal_slot = yes		# reconstruct THIS offer's globals from the buyer's slot
	if = {
		limit = {
			exists = event_target:sm_sale_offer_fleet
			exists = event_target:sm_sale_offer_seller
			event_target:sm_sale_offer_fleet = {
				has_fleet_flag = sm_offer_pending
				exists = owner
				NOT = { owner = { is_same_value = root } }
			}
			resource_stockpile_compare = { resource = trade value >= value:sm_sale_offer_price }
		}
		set_variable = { which = sm_last_market_value value = value:sm_sale_offer_price }
		add_resource = { trade = -1 mult = value:sm_sale_offer_price }
		event_target:sm_sale_offer_seller = {
			set_variable = { which = sm_last_market_value value = value:sm_sale_offer_price }
			add_resource = { trade = 1 mult = value:sm_sale_offer_price }
		}
		event_target:sm_sale_offer_fleet = {
			# Drop the reservation, then force off-mothball + 10-day recommission,
			# then transfer ownership and command to the buyer.
			sm_clear_sale_offer_fleet = yes
			sm_force_reactivate_fleet = yes
			set_owner = root
			set_controller = root
			clear_orders = yes
			if = {
				limit = { exists = root.capital_scope }
				set_location = { target = root.capital_scope distance = 0 }
			}
		}
		event_target:sm_sale_offer_seller = { sm_refresh_seller_mothball_modifier = yes }
		# Trust/opinion payoff for a completed sale (scales with existing relations).
		# v0.9.30 EXPLOIT FIX: only deals worth >= 300 trade credits build goodwill -
		# spamming corvette micro-sales was a trust/opinion farm. (was 1000 at
		# energy scale; rescaled x1/3 in v0.9.38)
		if = {
			limit = { check_variable = { which = sm_last_market_value value >= 300 } }
			sm_apply_trade_goodwill = { BUYER = sm_sale_offer_buyer  SELLER = sm_sale_offer_seller }
		}
		log = "SM_DEAL,direct_sale,buyer=[Root.GetName],seller=[sm_sale_offer_seller.GetName],price=[Root.GetShipMarketLastValue]"
		set_country_flag = sm_sale_bought
	}
	sm_free_deal_slot = yes		# release the slot + drop the "O" row (answered)
}

# COUNTRY SCOPE (buyer, root). Decline: release the seller's reserved fleet (back to
# plain mothball) and let the caller notify the seller.
sm_decline_sale_offer = {
	sm_load_deal_slot = yes		# reconstruct this offer's globals from the buyer's slot
	if = {
		limit = { exists = event_target:sm_sale_offer_fleet }
		event_target:sm_sale_offer_fleet = { sm_clear_sale_offer_fleet = yes }
	}
	sm_free_deal_slot = yes		# release the slot + drop the "O" row (answered)
}

# COUNTRY SCOPE (root = the RECIPIENT holding a pending offer). Monthly TTL tick:
# an offer left unanswered for ~30 days (sm_offer_ttl, ticked -30/month) is auto-
# declined - the fleet reservation is released, the sender is notified (declined),
# and the slot is freed. Fired by sm_offer.900 from on_monthly_pulse_country. This
# also reclaims slots from offers that never resolved (keeps the 6 slots from leaking).
# Type: sm_my_deal_type 1 = sale, 2 = lease. Depth-safe (shallow sub-calls).
sm_expire_stale_offers = {
	if = {
		limit = { has_country_flag = sm_pending_player_offer }
		change_variable = { which = sm_offer_ttl value = -30 }
		if = {
			limit = { check_variable = { which = sm_offer_ttl value < 0 } }
			sm_load_deal_slot = yes		# reconstruct THIS offer's globals from the recipient's slot
			if = {
				limit = { check_variable = { which = sm_my_deal_type value = 1 } }		# SALE
				if = { limit = { exists = event_target:sm_sale_offer_fleet } event_target:sm_sale_offer_fleet = { sm_clear_sale_offer_fleet = yes } }
				if = { limit = { exists = event_target:sm_sale_offer_seller } event_target:sm_sale_offer_seller = { country_event = { id = sm_offer.210 } } }
			}
			else_if = {
				limit = { check_variable = { which = sm_my_deal_type value = 2 } }		# LEASE
				if = { limit = { exists = event_target:sm_lease_offer_lessor } event_target:sm_lease_offer_lessor = { country_event = { id = sm_lease.56 } } }
			}
			clear_variable = sm_offer_ttl
			sm_free_deal_slot = yes		# release the slot + drop the "O" row
		}
	}
}

# COUNTRY SCOPE (seller, root). Withdraw every pending sale offer I have out — clears
# the reservation on each (fleets stay mothballed). For the manage menu.
sm_withdraw_my_sale_offers = {
	every_owned_fleet = {
		limit = { has_fleet_flag = sm_offer_pending }
		sm_clear_sale_offer_fleet = yes
	}
	# Release every concurrent deal slot + clear all "O" rows (sale AND lease pending).
	sm_free_all_deal_slots = yes
}

# ----- Trust / goodwill on a completed deal ----------------------------------
# ANY SCOPE. $BUYER$ / $SELLER$ = global event-target names for the two parties of a
# completed fleet deal (sale or lease). Builds MUTUAL trust + opinion. The buyer's
# gain is keyed on how much they ALREADY regard the seller ("based upon what they
# think of your empire"): a buyer who already likes you (opinion >= 40) deepens the
# partnership more (the _trusted modifier + more trust); otherwise the baseline.
# add_trust/add_opinion_modifier are engine effects (no scripted-effect depth added).
sm_apply_trade_goodwill = {
	event_target:$BUYER$ = {
		if = {
			limit = {
				exists = event_target:$SELLER$
				opinion = { who = event_target:$SELLER$  value >= 40 }
			}
			add_opinion_modifier = { who = event_target:$SELLER$  modifier = opinion_sm_arms_supplier_trusted }
			add_trust = { who = event_target:$SELLER$  amount = 15 }
		}
		else_if = {
			limit = { exists = event_target:$SELLER$ }
			add_opinion_modifier = { who = event_target:$SELLER$  modifier = opinion_sm_arms_supplier }
			add_trust = { who = event_target:$SELLER$  amount = 8 }
		}
	}
	event_target:$SELLER$ = {
		if = {
			limit = { exists = event_target:$BUYER$ }
			add_opinion_modifier = { who = event_target:$BUYER$  modifier = opinion_sm_arms_customer }
			add_trust = { who = event_target:$BUYER$  amount = 8 }
		}
	}
}
