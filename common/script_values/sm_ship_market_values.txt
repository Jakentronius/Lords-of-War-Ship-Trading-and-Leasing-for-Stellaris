# ----- Alloy build-cost estimate (RETAINED, not used for pricing) ------------

# -> alloy -> market chain produced 0 in testing. Safe to revisit later.

sm_alloys_corv = { base = 0  add = event_target:sm_stat_fleet.sm_n_corv  mult = 60 }
sm_alloys_frig = { base = 0  add = event_target:sm_stat_fleet.sm_n_frig  mult = 90 }
sm_alloys_dest = { base = 0  add = event_target:sm_stat_fleet.sm_n_dest  mult = 120 }

sm_alloys_tit  = { base = 0  add = event_target:sm_stat_fleet.sm_n_tit   mult = 2000 }
sm_fleet_alloys = {
	base = 0
	# value: prefix required - a bare name here is read as a VARIABLE on the scope
	# (not set -> 0), which silently zeroed the whole alloy cost. (fixed v0.9.9)
	add = value:sm_alloys_corv
	add = value:sm_alloys_frig
	add = value:sm_alloys_dest
	add = value:sm_alloys_cru
	add = value:sm_alloys_bb
	add = value:sm_alloys_tit

# LIVE-MARKET APPRAISAL (v0.9.39): the trade cost of buying this fleet's alloy
# content on the market RIGHT NOW (amount = the fleet's stored sm_cost_alloys,
# trade_type = market_buy - vanilla sell_resource_at_market_value pattern).
# Evaluate in the owner's (country/market) scope with sm_stat_fleet saved.
# Returns 0 when unavailable -> callers fall back to the static table below.
sm_fleet_market_value = {
	base = 0
	complex_trigger_modifier = {
		trigger = market_resource_price
		parameters = { resource = alloys  amount = event_target:sm_stat_fleet.sm_cost_alloys  trade_type = market_buy }
		mode = add
	}
}

# Estimated TRADE build cost of the fleet, counted at mothball time. Uses the engine
# counter count_owned_ship (a complex_trigger_modifier — NOT the manual every_owned_ship
# loop that produced 0) weighted per ship size by a tunable trade-credit-equivalent of its
# build resources. The count is restricted to THIS fleet via fleet=event_target:sm_stat_fleet,
# so it is correct regardless of count_owned_ship's default scope. Evaluate in FLEET scope
# at mothball and store as sm_build_value (see sm_capture_fleet_stats).
# TRADE SCALE (v0.9.39, x2 from v0.9.38): FALLBACK table only - the primary
# appraisal is now the LIVE MARKET (sm_apply_market_value: alloy content at the
# current market_buy price, ~4 trade/alloy base). This static table sits ~2x
# below the market anchor; a 30-battleship doomstack ~= 30000 (sells ~45000,
# right under the 50k trade stockpile cap).
sm_fleet_build_value = {
	base = 0
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 80   parameters = { limit = { is_ship_size = corvette   fleet = { is_same_value = event_target:sm_stat_fleet } } } }
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 120  parameters = { limit = { is_ship_size = frigate    fleet = { is_same_value = event_target:sm_stat_fleet } } } }
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 170  parameters = { limit = { is_ship_size = destroyer  fleet = { is_same_value = event_target:sm_stat_fleet } } } }
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 400  parameters = { limit = { is_ship_size = cruiser    fleet = { is_same_value = event_target:sm_stat_fleet } } } }
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 1000 parameters = { limit = { is_ship_size = battleship fleet = { is_same_value = event_target:sm_stat_fleet } } } }
	complex_trigger_modifier = { trigger = count_owned_ship  mode = add  mult = 3300 parameters = { limit = { is_ship_size = titan      fleet = { is_same_value = event_target:sm_stat_fleet } } } }
}

# Suggested TRADE sale price = estimated build cost x1.5 (a markup over cost). Reads the

# back to the robust fleet-power valuation, so a price is never 0.
sm_recommended_price = { base = 0  add = event_target:sm_pending_fleet.sm_build_value  mult = 1.5 }
